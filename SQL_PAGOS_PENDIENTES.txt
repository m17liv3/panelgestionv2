SQL para activar pagos pendientes en renovaciones

Ejecuta este SQL una sola vez en Supabase > SQL Editor.

alter table public.renovaciones
add column if not exists estado_pago text default 'pagado',
add column if not exists fecha_pago timestamptz;

update public.renovaciones
set estado_pago = 'pagado'
where estado_pago is null;
