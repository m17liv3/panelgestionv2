-- Añade estado de contestación al aviso de renovación.
-- Ejecutar una sola vez en Supabase SQL Editor.

alter table public.clientes
add column if not exists aviso_renovacion_contestado boolean default false,
add column if not exists aviso_renovacion_contestado_fecha timestamptz,
add column if not exists aviso_renovacion_contestado_expiracion date;

update public.clientes
set aviso_renovacion_contestado = false
where aviso_renovacion_contestado is null;
