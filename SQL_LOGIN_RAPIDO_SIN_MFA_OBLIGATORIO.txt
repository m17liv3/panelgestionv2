-- M17LIV3 - Login rápido sin Google Authenticator obligatorio
-- Ejecutar UNA SOLA VEZ en Supabase SQL Editor.
-- Mantiene seguridad por usuario autenticado: cada usuario solo ve sus propios datos.
-- Quita el requisito aal2 de Google Authenticator en las políticas RLS.

-- CLIENTES
drop policy if exists "clientes_select_owner" on public.clientes;
drop policy if exists "clientes_insert_owner" on public.clientes;
drop policy if exists "clientes_update_owner" on public.clientes;
drop policy if exists "clientes_delete_owner" on public.clientes;

create policy "clientes_select_owner"
on public.clientes
for select
to authenticated
using (owner_id = (select auth.uid()));

create policy "clientes_insert_owner"
on public.clientes
for insert
to authenticated
with check (owner_id = (select auth.uid()));

create policy "clientes_update_owner"
on public.clientes
for update
to authenticated
using (owner_id = (select auth.uid()))
with check (owner_id = (select auth.uid()));

create policy "clientes_delete_owner"
on public.clientes
for delete
to authenticated
using (owner_id = (select auth.uid()));
