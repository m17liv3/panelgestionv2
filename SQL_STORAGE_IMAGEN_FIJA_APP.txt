-- Crear bucket público para la imagen fija de la app Android.
-- Ejecutar una sola vez en Supabase > SQL Editor.
-- Después, desde la web-app podrás subir/sustituir siempre la misma imagen.

insert into storage.buckets (id, name, public, file_size_limit, allowed_mime_types)
values (
  'cartelera',
  'cartelera',
  true,
  10485760,
  array['image/jpeg','image/png','image/webp','image/gif']::text[]
)
on conflict (id) do update
set public = true,
    file_size_limit = excluded.file_size_limit,
    allowed_mime_types = excluded.allowed_mime_types;

-- Lectura pública para que la app Android pueda ver la imagen sin iniciar sesión.
drop policy if exists "cartelera_public_read" on storage.objects;
create policy "cartelera_public_read"
on storage.objects
for select
to public
using (bucket_id = 'cartelera');

-- Solo usuarios autenticados con Google Authenticator/MFA pueden subir la imagen fija.
drop policy if exists "cartelera_mfa_insert" on storage.objects;
create policy "cartelera_mfa_insert"
on storage.objects
for insert
to authenticated
with check (
  bucket_id = 'cartelera'
  and name = 'imagen_actual.jpg'
  and (select auth.jwt()->>'aal') = 'aal2'
);

-- Permite sustituir la imagen manteniendo el mismo link.
drop policy if exists "cartelera_mfa_update" on storage.objects;
create policy "cartelera_mfa_update"
on storage.objects
for update
to authenticated
using (
  bucket_id = 'cartelera'
  and name = 'imagen_actual.jpg'
  and (select auth.jwt()->>'aal') = 'aal2'
)
with check (
  bucket_id = 'cartelera'
  and name = 'imagen_actual.jpg'
  and (select auth.jwt()->>'aal') = 'aal2'
);

-- Permite borrar la imagen desde la web-app si lo necesitas.
drop policy if exists "cartelera_mfa_delete" on storage.objects;
create policy "cartelera_mfa_delete"
on storage.objects
for delete
to authenticated
using (
  bucket_id = 'cartelera'
  and name = 'imagen_actual.jpg'
  and (select auth.jwt()->>'aal') = 'aal2'
);
